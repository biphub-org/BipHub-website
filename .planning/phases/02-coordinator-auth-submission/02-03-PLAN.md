```typescript
import { NextResponse, type NextRequest } from 'next/server'
import { createMiddlewareClient } from '@/lib/supabase/middleware'

/**
 * Edge middleware for BipHub (Phase 2).
 *
 * Responsibilities:
 *   1. Refresh the Supabase session cookie on every matched request via getClaims()
 *      — getClaims validates the JWT signature locally (PITFALLS Pitfall 1).
 *   2. Inject `x-pathname` response header so RSC layouts (notably the
 *      (dashboard) layout's profile-complete gate) can read the current path
 *      without parsing referer (Pitfall 2 prevention).
 *   3. Phase 2 redirects:
 *        - !claims && pathname.startsWith('/dashboard' | '/onboarding') → /login
 *        - claims && pathname === '/login' | '/register' → /dashboard
 *
 * Phase 3 will add an admin role guard on /admin paths.
 *
 * NEVER use getSession() — it does not validate JWT signatures.
 */
export async function middleware(request: NextRequest) {
  const { supabase, response } = createMiddlewareClient(request)

  // CRITICAL: getClaims() validates the JWT signature on every request.
  const { data: { claims } } = await supabase.auth.getClaims()

  const { pathname } = request.nextUrl

  // (2) Inject pathname header for downstream RSC layouts (Pitfall 2 fix).
  response.headers.set('x-pathname', pathname)

  // (3a) Auth-required: dashboard + onboarding.
  if (pathname.startsWith('/dashboard') || pathname.startsWith('/onboarding')) {
    if (!claims) {
      return NextResponse.redirect(new URL('/login', request.url))
    }
  }

  // (3b) Already-authenticated: bounce off the auth pages.
  // Note: matcher excludes /login and /register from middleware execution by default
  // (existing config), so this branch only fires if the matcher is later expanded.
  // Kept here for defense-in-depth and clarity if matcher changes in Phase 3+.
  if (claims && (pathname === '/login' || pathname === '/register')) {
    return NextResponse.redirect(new URL('/dashboard', request.url))
  }

  return response
}

// Matcher set in Plan 01-01. DO NOT modify.
export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|login|register|auth|.*\\.(?:svg|png|jpg|jpeg|gif|webp|json)$).*)',
  ],
}
```

**2. Convert `app/(public)/layout.tsx` to async RSC:**

```typescript
import { StickyNav } from '@/components/home/StickyNav'
import { Footer } from '@/components/home/Footer'
import { Toaster } from '@/components/ui/sonner'
import { createClient } from '@/lib/supabase/server'

/**
 * Public route-group layout.
 *
 * Phase 2 addition (D-15): fetch session claims server-side via getClaims() and
 * pass `hasClaims` + `initials` to <StickyNav> so the nav adapts without a
 * client-side flash.
 *
 * Initials derivation:
 *   1. profiles.full_name → first letter of first two words.
 *   2. fallback to first two chars of email local part.
 *   3. "··" if neither is available (defensive).
 */
export default async function PublicLayout({
  children,
}: {
  children: React.ReactNode
}) {
  const supabase = await createClient()
  const { data: { claims } } = await supabase.auth.getClaims()

  let initials: string | null = null
  const hasClaims = Boolean(claims)

  if (claims?.sub) {
    // Profile name first; fall back to email.
    const { data: profile } = await supabase
      .from('profiles')
      .select('full_name')
      .eq('id', claims.sub)
      .maybeSingle()

    const fromName = profile?.full_name
      ? profile.full_name
          .trim()
          .split(/\s+/)
          .slice(0, 2)
          .map((w) => w[0])
          .join('')
          .toUpperCase()
      : null

    const emailLocal = typeof claims.email === 'string' ? claims.email.split('@')[0] : null
    const fromEmail = emailLocal ? emailLocal.slice(0, 2).toUpperCase() : null

    initials = fromName || fromEmail || '··'
  }

  return (
    <>
      <a href="#main" className="skip-link">Skip to main content</a>
      <StickyNav hasClaims={hasClaims} initials={initials} />
      <main id="main" className="min-h-[calc(100vh-68px)]">
        {children}
      </main>
      <Footer />
      <Toaster
        position="bottom-right"
        richColors={false}
        closeButton
      />
    </>
  )
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → middleware | JWT cookie crosses every matched request; validated by `getClaims()` |
| middleware → RSC layout | `x-pathname` header is server-set on response; RSC layouts trust it |
| RSC layout → client component | `hasClaims` + `initials` are serialized via React props (no client-side fetch, no flash) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Spoofing | Middleware trusts unvalidated session | mitigate | `getClaims()` validates JWT signature; `getSession()` is never called (CLAUDE.md never-do) |
| T-02-03-02 | Tampering | Client forges `x-pathname` to fool dashboard layout | mitigate | Header is set on the server-controlled response by middleware; Next.js incoming-request headers do not propagate `x-pathname` to RSC `headers()` reads — only the response.headers we set are visible. RSC reads from `headers()` which returns the *request* headers, but Next.js exposes the headers we set via the middleware response. Verified pattern from RESEARCH.md A1 / Pitfall 2; cross-check during integration in Plan 02-04. |
| T-02-03-03 | Information Disclosure | `getClaims()` in public layout exposes profile.full_name to all renders | accept | full_name is intended for display in nav; not sensitive PII beyond the user's own session |
| T-02-03-04 | Denial of Service | Adding a profile DB query to every public render adds latency | accept | Single indexed select on PK (`profiles.id = claims.sub`); negligible cost at <1k users (research doc Pattern 9) |
| T-02-03-05 | Elevation of Privilege | Auth-required path bypassed when matcher misses | mitigate | `pathname.startsWith('/dashboard') || pathname.startsWith('/onboarding')` matches every nested route; the matcher excludes only `/login`, `/register`, `/auth`, static assets — `/dashboard*` is never excluded |
| T-02-03-06 | Elevation of Privilege | Open redirect via crafted nextUrl | mitigate | Redirect targets are hard-coded literals (`'/login'`, `'/dashboard'`); no user input reaches the URL constructor |
| T-02-03-07 | Tampering | Race between layout's getClaims and middleware refresh | accept | Both run in the same request; `(public)/layout.tsx` reads after middleware has set the cookie via the Edge runtime cookie protocol |
</threat_model>
