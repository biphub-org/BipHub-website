    export async function adminUpdateBipAction(bipId: string, data: BipDraftData): Promise<ActionResult> {
      const supabase = await createClient()
      const { data: authData, error: authError } = await supabase.auth.getClaims()
      const claims = authData?.claims ?? null
      if (authError || !claims?.sub) return { error: 'Your session has expired. Please sign in again.' }
      const role = (claims as { app_metadata?: { role?: string } }).app_metadata?.role
      if (role !== 'admin') return { error: 'Forbidden.' }

      if (!bipId || typeof bipId !== 'string') return { error: 'Invalid BIP id.' }

      // Read pre-image to determine revalidatePath behavior + audit row from_status
      const { data: existing, error: readError } = await supabase
        .from('bips')
        .select('id, slug, title, status')
        .eq('id', bipId)
        .maybeSingle()
      if (readError || !existing) return { error: 'BIP not found.' }

      // Validate the wizard data using the same Zod schemas as submitBipAction.
      // Reuse Phase 2's BipDraftSchema or per-step schemas — read
      // lib/schemas/bip-wizard.ts for the canonical validator. If a single
      // "full draft" schema exists, use it. Otherwise concatenate step schemas.
      // EXECUTOR: inspect bip-submit.ts to identify the validator used there;
      // mirror it here exactly.
      // const parsed = BipDraftSchema.safeParse(data)
      // if (!parsed.success) return { error: parsed.error.issues[0]?.message ?? 'Invalid input.' }

      // Map wizard data → bips columns. The mapping is identical to
      // submitBipAction's `finalize` step EXCEPT:
      //   - Do NOT change status (preserve existing.status)
      //   - Do NOT call validateTransition (no transition occurs)
      //   - Do NOT call finalizeSlug (admin edits to an already-public BIP
      //     should not change the slug — slug-stability for ISR)
      //   - Partner upsert (bip_partner_universities) still runs

      const updatePayload: Record<string, unknown> = {
        // Map all editable wizard fields to bips columns here.
        // Use the same field list as submitBipAction's final bips update.
        // EXAMPLE (executor must extract the real list from bip-submit.ts):

        description: data.description,
        learning_outcomes: data.learning_outcomes,
        // ...all other editable columns...
        updated_at: new Date().toISOString(),
        // status: NOT included — D-18
      }

      const { error: updateError } = await supabase
        .from('bips')
        .update(updatePayload)
        .eq('id', bipId)
      if (updateError) {
        console.error('[adminUpdateBipAction] update error:', updateError.message)
        return { error: 'Failed to save changes. Please try again.' }
      }

      // Partner upsert — mirror submitBipAction's partner handling block.
      // (Copy the partner upsert logic verbatim from bip-submit.ts, but skip
      //  any status-related steps.)

      // Audit row
      const { error: auditError } = await supabase.from('bip_status_history').insert({

        from_status: existing.status,
        to_status: existing.status,  // unchanged
        actor_id: claims.sub,
        note: `Admin edit by ${claims.email ?? claims.sub}`,

    export function AdminEditFooter({ bipId, bipTitle, coordinatorName, currentStatus, data }: Props) {

            </Button>
          </div>

            open={approveOpen}
            onOpenChange={setApproveOpen}
            bipId={bipId}
            bipTitle={bipTitle}
            coordinatorName={coordinatorName}
          />
          <RejectBipModal

            onOpenChange={setRejectOpen}
