   ```typescript
   /**
    * BipHub logo mark — 32×32 navy square with a gold star ring.
    *
    * !!! LEGAL CONSTRAINT — DO NOT MODIFY WITHOUT READING:
    *   - PITFALLS.md Pitfall 8 (EU 12-star emblem)
    *   - 01-UI-SPEC.md line 122
    *   - .planning/STATE.md (open blocker)
    *
    * The star count MUST NOT be 12. Using exactly 11 stars in a circular
    * arrangement, which is visually distinct from the EU emblem.
    * Plan 01-04 enforces this; Phase 4 ships the CONTRIBUTING.md note.
    */

   const STAR_COUNT = 11 // NEVER 12 — see file header

   export function LogoMark({ className }: { className?: string }) {
     const radius = 13
     const center = 16
     const dotRadius = 1.25
     const stars = Array.from({ length: STAR_COUNT }, (_, i) => {
       const angle = (2 * Math.PI * i) / STAR_COUNT - Math.PI / 2
       return {
         cx: center + radius * Math.cos(angle),
         cy: center + radius * Math.sin(angle),
       }
     })

     return (
       <svg
         viewBox="0 0 32 32"
         className={className}
         width={32}
         height={32}
         aria-hidden="true"
         role="presentation"
       >
         <rect x={0} y={0} width={32} height={32} rx={8} className="fill-eu-blue" />
         {stars.map((s, i) => (
           <circle
             key={i}
             cx={s.cx}
             cy={s.cy}
             r={dotRadius}
             className="fill-eu-gold"
           />
         ))}
         <text
           x={16}
           y={20}
           textAnchor="middle"
           className="fill-white font-bold"
           style={{ fontSize: 12 }}
         >
           B
         </text>
       </svg>
     )
   }
   ```
